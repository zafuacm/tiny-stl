174// 红黑树迭代器
175template <class T>
176struct rb_tree_iterator_base
177 : public tstl::iterator<tstl::bidirectional_iterator_tag, T> { //继承自双向迭代器
178 using base_ptr = typename rb_tree_traits<T>::base_ptr;
179
180 base_ptr node; // 指向节点本身
181
182 rb_tree_iterator_base() : node(nullptr) {
183 }
184
185 // 迭代器的后继
186 void inc() {
187 if (node->right != nullptr) {
188 node = rb_tree_min(node->right);
189 } else { // 如果没有右子节点
190 auto y = node->parent;
191 while (y->right == node) {
192 node = y;
193 y = y->parent;
194 }
195 // 应对“寻找根节点的下一节点，而根节点没有右子节点”的特殊情况
196 if (node->right != y)
197 node = y;
198 }
199 }
200
201 // 迭代器的前驱
202 void dec() {
203 if (node->parent->parent == node && rb_tree_is_red(node)) { // 如果 node 为 header
204 node = node->right; // 指向整棵树的 max 节点
205 } else if (node->left != nullptr) {
206 node = rb_tree_max(node->left);
207 } else { // 非 header 节点，也无左子节点
208 auto y = node->parent;
209 while (node == y->left) {
210 node = y;
211 y = y->parent;
212 }
213 node = y;
214 }
215 }
216
217 bool operator==(const rb_tree_iterator_base &rhs) {
218 return node == rhs.node;
219 }
220 bool operator!=(const rb_tree_iterator_base &rhs) {
221 return node != rhs.node;
222 }
223};

388// 左旋，（左旋点，根节点）
389template <class NodePtr>
390void rb_tree_rotate_left(NodePtr x, NodePtr &root) noexcept {
391 auto y = x->right; // y 为 x 的右子节点
392 x->right = y->left;
393 if (y->left != nullptr) {
394 y->left->parent = x;
395 }
396 y->parent = x->parent;
397
398 if (x == root) { // 如果 x 为根节点，让 y 顶替 x 成为根节点
399 root = y;
400 } else if (rb_tree_is_lchild(x)) { // 如果 x 是左子节点
401 x->parent->left = y;
402 } else { // 如果 x 是右子节点
403 x->parent->right = y;
404 }
405 // 调整 x 与 y 的关系
406 y->left = x;
407 x->parent = y;
408}

432// 插入节点后使 rb tree 重新平衡，参数一为新增节点，参数二为根节点
433//
434// case 1: 新增节点位于根节点，令新增节点为黑
435// case 2: 新增节点的父节点为黑，没有破坏平衡，直接返回
436// case 3: 父节点和叔叔节点都为红，令父节点和叔叔节点为黑，祖父节点为红，
437// 然后令祖父节点为当前节点，继续处理
438// case 4: 父节点为红，叔叔节点为 NIL
439// 或黑色，父节点为左（右）孩子，当前节点为右（左）孩子，
440// 让父节点成为当前节点，再以当前节点为支点左（右）旋
441// case 5: 父节点为红，叔叔节点为 NIL
442// 或黑色，父节点为左（右）孩子，当前节点为左（右）孩子，
443// 让父节点变为黑色，祖父节点变为红色，以祖父节点为支点右（左）旋
444// 插入平衡函数
445template <class NodePtr>
446void rb_tree_insert_rebalance(NodePtr x, NodePtr &root) noexcept {
447 rb_tree_set_red(x); // 新增节点为红色
448 while (x != root && rb_tree_is_red(x->parent)) {
449 if (rb_tree_is_lchild(x->parent)) { // 如果父节点是左子节点
450 auto uncle = x->parent->parent->right;
451 if (uncle != nullptr && rb_tree_is_red(uncle)) { // case 3: 父节点和叔叔节点都为红
452 rb_tree_set_black(x->parent);
453 rb_tree_set_black(uncle);
454 x = x->parent->parent;
455 rb_tree_set_red(x);
456 } else { // 无叔叔节点或叔叔节点为黑
457 if (!rb_tree_is_lchild(x)) { // case 4: 当前节点 x 为右子节点
458 x = x->parent;
459 rb_tree_rotate_left(x, root);
460 }
461 // 都转换成 case 5： 当前节点为左子节点
462 rb_tree_set_black(x->parent);
463 rb_tree_set_red(x->parent->parent);
464 rb_tree_rotate_right(x->parent->parent, root);
465 break;
466 }
467 } else { // 如果父节点是右子节点，对称处理
468 auto uncle = x->parent->parent->left;
469 if (uncle != nullptr && rb_tree_is_red(uncle)) { // case 3: 父节点和叔叔节点都为红
470 rb_tree_set_black(x->parent);
471 rb_tree_set_black(uncle);
472 x = x->parent->parent;
473 rb_tree_set_red(x);
474 // 此时祖父节点为红，可能会破坏红黑树的性质，令当前节点为祖父节点，继续处理
475 } else { // 无叔叔节点或叔叔节点为黑
476 if (rb_tree_is_lchild(x)) { // case 4: 当前节点 x 为左子节点
477 x = x->parent;
478 rb_tree_rotate_right(x, root);
479 }
480 // 都转换成 case 5： 当前节点为左子节点
481 rb_tree_set_black(x->parent);
482 rb_tree_set_red(x->parent->parent);
483 rb_tree_rotate_left(x->parent->parent, root);
484 break;
485 }
486 }
487 }
488 rb_tree_set_black(root); // 根节点永远为黑
489}

491// 删除节点后的自平衡
492template <class NodePtr>
493NodePtr rb_tree_erase_rebalance(NodePtr z, NodePtr &root, NodePtr &leftmost, NodePtr &rightmost) {
494 // y 是可能的替换节点，指向最终要删除的节点
495 auto y = (z->left == nullptr || z->right == nullptr) ? z : rb_tree_next(z);
496 // x 是 y 的一个独子节点或 NIL 节点
497 auto x = y->left != nullptr ? y->left : y->right;
498 // xp 为 x 的父节点
499 NodePtr xp = nullptr;
500
501 // y != z 说明 z 有两个非空子节点，此时 y 指向 z 右子树的最左节点，x 指向 y
502 // 的右子节点。 用 y 顶替 z 的位置，用 x 顶替 y 的位置，最后用 y 指向 z
503 if (y != z) {
504 z->left->parent = y;
505 y->left = z->left;
506
507 // 如果 y 不是 z 的右子节点，那么 z 的右子节点一定有左孩子
508 if (y != z->right) { // x 替换 y 的位置
509 xp = y->parent;
510 if (x != nullptr) {
511 x->parent = y->parent;
512 }
513
514 y->parent->left = x;
515 y->right = z->right;
516 z->right->parent = y;
517 } else {
518 xp = y;
519 }
520
521 // 连接 y 与 z 的父节点
522 if (root == z) {
523 root = y;
524 } else if (rb_tree_is_lchild(z)) {
525 z->parent->left = y;
526 } else {
527 z->parent->right = y;
528 }
529 y->parent = z->parent;
530 tstl::swap(y->color, z->color);
531 y = z;
532 }
533 // y == z 说明 z 至多只有一个孩子
534 else {
535 xp = y->parent;
536 if (x) {
537 x->parent = y->parent;
538 }
539
540 // 连接 x 与 z 的父节点
541 if (root == z) {
542 root = x;
543 } else if (rb_tree_is_lchild(z)) {
544 z->parent->left = x;
545 } else {
546 z->parent->right = x;
547 }
548
549 // 此时 z 有可能是最左节点或最右节点，更新数据
550 if (leftmost == z) {
551 leftmost = x == nullptr ? xp : rb_tree_min(x);
552 }
553 if (rightmost == z) {
554 rightmost = x == nullptr ? xp : rb_tree_max(x);
555 }
556 }
557
558 // 此时，y 指向要删除的节点，x 为替代节点，从 x 节点开始调整。
559 // 如果删除的节点为红色，树的性质没有被破坏，否则按照以下情况调整（x
560 // 为左子节点为例）： case 1:
561 // 兄弟节点为红色，令父节点为红，兄弟节点为黑，进行左（右）旋，继续处理 case
562 // 2: 兄弟节点为黑色，且两个子节点都为黑色或
563 // NIL，令兄弟节点为红，父节点成为当前节点，继续处理 case 3:
564 // 兄弟节点为黑色，左子节点为红色或 NIL，右子节点为黑色或 NIL，
565 // 令兄弟节点为红，兄弟节点的左子节点为黑，以兄弟节点为支点右（左）旋，继续处理
566 // case 4:
567 // 兄弟节点为黑色，右子节点为红色，令兄弟节点为父节点的颜色，父节点为黑色，兄弟节点的右子节点
568 // 为黑色，以父节点为支点左（右）旋，树的性质调整完成，算法结束
569 if (!rb_tree_is_red(y)) { // x 为黑色时，调整，否则直接将 x 变为黑色即可
570 while (x != root && (x == nullptr || !rb_tree_is_red(x))) {
571 if (x == xp->left) { // 如果 x 为左子节点
572 auto brother = xp->right;
573 if (rb_tree_is_red(brother)) { // case 1
574 rb_tree_set_black(brother);
575 rb_tree_set_red(xp);
576 rb_tree_rotate_left(xp, root);
577 brother = xp->right;
578 }
579 // case 1 转为为了 case 2、3、4 中的一种
580 if ((brother->left == nullptr || !rb_tree_is_red(brother->left)) &&
581 (brother->right == nullptr || !rb_tree_is_red(brother->right))) { // case 2
582 rb_tree_set_red(brother);
583 x = xp;
584 xp = xp->parent;
585 } else {
586 if (brother->right == nullptr || !rb_tree_is_red(brother->right)) { // case 3
587 if (brother->left != nullptr) {
588 rb_tree_set_black(brother->left);
589 }
590 rb_tree_set_red(brother);
591 rb_tree_rotate_right(brother, root);
592 brother = xp->right;
593 }
594 // 转为 case 4
595 brother->color = xp->color;
596 rb_tree_set_black(xp);
597 if (brother->right != nullptr) {
598 rb_tree_set_black(brother->right);
599 }
600 rb_tree_rotate_left(xp, root);
601 break;
602 }
603 } else { // x 为右子节点，对称处理
604 auto brother = xp->left;
605 if (rb_tree_is_red(brother)) { // case 1
606 rb_tree_set_black(brother);
607 rb_tree_set_red(xp);
608 rb_tree_rotate_right(xp, root);
609 brother = xp->left;
610 }
611 if ((brother->left == nullptr || !rb_tree_is_red(brother->left)) &&
612 (brother->right == nullptr || !rb_tree_is_red(brother->right))) { // case 2
613 rb_tree_set_red(brother);
614 x = xp;
615 xp = xp->parent;
616 } else {
617 if (brother->left == nullptr || !rb_tree_is_red(brother->left)) { // case 3
618 if (brother->right != nullptr) {
619 rb_tree_set_black(brother->right);
620 }
621 rb_tree_set_red(brother);
622 rb_tree_rotate_left(brother, root);
623 brother = xp->left;
624 }
625 // 转为 case 4
626 brother->color = xp->color;
627 rb_tree_set_black(xp);
628 if (brother->left != nullptr) {
629 rb_tree_set_black(brother->left);
630 }
631 rb_tree_rotate_right(xp, root);
632 break;
633 }
634 }
635 }
636 if (x != nullptr) {
637 rb_tree_set_black(x);
638 }
639 }
640 return y;
641}

1159 std::pair<std::pair<base_ptr, bool>, bool> get_insert_unique_pos(const key_type &key) {
1160 // 返回一个 pair，第一个值为一个 pair，包含插入点的父节点和一个
1161 // bool 表示是否在左边插入，
1162 // 第二个值为一个 bool，表示是否插入成功
1163 auto x = root();
1164 auto y = m_header;
1165 bool add_to_left = true; // 树为空时也在 header_ 左边插入
1166 while (x != nullptr) {
1167 y = x;
1168 add_to_left = m_key_comp(key, value_traits::get_key(x->get_node_ptr()->value));
1169 x = add_to_left ? x->left : x->right;
1170 }
1171 iterator j = iterator(y); // 此时 y 为插入点的父节点
1172 if (add_to_left) {
1173 if (y == m_header ||
1174 j == begin()) { // 如果树为空树或插入点在最左节点处，肯定可以插入新的节点
1175 return std::make_pair(std::make_pair(y, true), true);
1176 } else { // 否则，如果存在重复节点，那么 --j 就是重复的值
1177 --j;
1178 }
1179 }
1180 if (m_key_comp(value_traits::get_key(*j), key)) { // 表明新节点没有重复
1181 return std::make_pair(std::make_pair(y, add_to_left), true);
1182 }
1183 // 进行至此，表示新节点与现有节点键值重复
1184 return std::make_pair(std::make_pair(y, add_to_left), false);
1185 }

1278 top->parent = p;
1279 try {
1280 if (x->right) {
1281 top->right = copy_from(x->right, top);

1288 y->parent = p;

1291 }
1292 p = y;
1293 x = x->left;
1294 }
