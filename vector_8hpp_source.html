547 friend bool operator==(const vector &lhs, const vector &rhs) {
548 if (lhs.size() != rhs.size()) {
549 return false;
550 }
551 auto lhs_i = lhs.begin(), rhs_i = rhs.begin();
552 while (lhs_i != lhs.end() && rhs_i != rhs.end()) {
553 if (*lhs_i != *rhs_i) {
554 return false;
555 }
556 lhs_i++, rhs_i++;
557 }
558 return true;

700 template <class ForwardIt>
701 void m_range_insert(iterator pos, ForwardIt first, ForwardIt last, tstl::forward_iterator_tag) {
702 if (first == last)
703 return;
704 const size_type n = tstl::distance(first, last);
705 if (size_type(m_end_of_storage - m_finish) >= n) {
706 const size_type elems_after = end() - pos;
707 pointer old_finish = m_finish;
708 if (elems_after > n) {
709 tstl::_uninitialized_move_a(m_finish - n, m_finish, m_finish, m_alloc);
710 m_finish += n;
711 tstl::move_backward(pos.base(), old_finish - n, old_finish);
712 tstl::copy(first, last, pos);
713 } else {
714 ForwardIt mid = first;
715 tstl::advance(mid, elems_after);
716 tstl::_uninitialized_copy_a(mid, last, m_finish, m_alloc);
717 m_finish += n - elems_after;
718 tstl::_uninitialized_move_a(pos.base(), old_finish, m_finish, m_alloc);
719 m_finish += elems_after;
720 tstl::copy(first, mid, pos);
721 }
722 } else {
723 const size_type len = m_check_len(n);
724 pointer new_start = m_allocate(len);
725 pointer new_finish = new_start;
726 try {
727 new_finish = tstl::_uninitialized_copy_a(m_start, pos.base(), new_start, m_alloc);
728 new_finish = tstl::_uninitialized_copy_a(first, last, new_finish, m_alloc);
729 new_finish = tstl::_uninitialized_copy_a(pos.base(), m_finish, new_finish, m_alloc);
730 } catch (...) {
731 m_destroy(new_start, new_finish);
732 m_deallocate(new_start, len);
733 throw;
734 }
735 m_destroy(m_start, m_finish);
736 m_deallocate(m_start, m_end_of_storage - m_start);
737 m_start = new_start;
738 m_finish = new_finish;
739 m_end_of_storage = new_start + len;
740 }
741 }

748 void m_fill_insert(iterator pos, size_type n, const T &value) {
749 if (n == 0)
750 return;
751 if (size_type(m_end_of_storage - m_finish) >= n) {
752 const size_type elems_after = end() - pos;
753 pointer old_finish = m_finish;
754 if (elems_after > n) {
755 tstl::_uninitialized_move_a(m_finish - n, m_finish, m_finish, m_alloc);
756 m_finish += n;
757 tstl::move_backward(pos.base(), old_finish - n, old_finish);
758 tstl::fill(pos.base(), pos.base() + n, value);
759 } else {
760 m_finish = tstl::_uninitialized_fill_n_a(m_finish, n - elems_after, value, m_alloc);
761 _uninitialized_move_a(pos.base(), old_finish, m_finish, m_alloc);
762 m_finish += elems_after;
763 tstl::fill(pos.base(), old_finish, value);
764 }
765 } else {
766 const size_type len = m_check_len(n);
767 const size_type elems_before = pos - begin();
768 pointer new_start = m_allocate(len);
769 pointer new_finish = new_start;
770 try {
771 tstl::_uninitialized_fill_n_a(new_start + elems_before, n, value, m_alloc);
772 new_finish = nullptr;
773 new_finish = tstl::_uninitialized_copy_a(m_start, pos.base(), new_start, m_alloc);
774 new_finish += n;
775
776 new_finish = tstl::_uninitialized_copy_a(pos.base(), m_finish, new_finish, m_alloc);
777
778 } catch (...) {
779 if (new_finish != nullptr) {
780 m_destroy(new_start + elems_before, new_start + elems_before + n);
781 } else {
782 m_destroy(new_start + elems_before + n);
783 }
784 m_deallocate(new_start, len);
785 throw;
786 }
787 m_destroy(m_start, m_finish);
788 m_deallocate(m_start, m_end_of_storage - m_start);
789 m_start = new_start;
790 m_finish = new_finish;
791 m_end_of_storage = new_start + len;
792 }

805 template <class... Args>
806 void m_realloc_insert(iterator pos, Args &&...args) {
807 const size_type len = m_check_len(1);
808 pointer old_start = m_start;
809 pointer old_finish = m_finish;
810 const size_type elems_before = pos - begin();
811 pointer new_start = m_allocate(len);
812 pointer new_finish = new_start;
813 try {
814 m_construct(new_start + elems_before, std::forward<Args>(args)...);
815 new_finish = pointer();
816 new_finish = m_uninitialized_copy(old_start, pos.base(), new_start);
817 ++new_finish;
818 new_finish = m_uninitialized_copy(pos.base(), old_finish, new_finish);
819 } catch (...) {
820 if (new_finish == nullptr) {
821 m_destroy(new_start, new_start + elems_before);
822 } else {
823 m_destroy(new_start, new_finish);
824 }
825 m_deallocate(new_start, len);
826 throw;
827 }
828 m_destroy(old_start, old_finish);
829 m_deallocate(old_start, m_end_of_storage - old_start);
830 m_start = new_start;
831 m_finish = new_finish;
832 m_end_of_storage = new_start + len;
833 }

889 void m_default_append(size_type n) {
890 if (n != 0) {
891 const size_type sz = size();
892 size_type navail = m_end_of_storage - m_finish;
893 if (navail >= n) {
894 m_finish = tstl::_uninitialized_default_construct_n_a(m_finish, n, m_alloc);
895 } else {
896 const size_type len = m_check_len(n);
897 pointer new_start = m_allocate(len);
898 pointer destroy_from = nullptr;
899 try {
