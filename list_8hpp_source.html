505 // 归并排序
506 void sort() {
507 if (m_node->next == m_node || m_node->next->next == m_node) {
508 return;
509 }
510 // 数据缓存区
511 // _tmp[n]中最多存放2^(n+1)个元素，若大于则与_tmp[n+1]作归并
512 list carry;
513 list counter[64];
514 int fill = 0, size = m_size;
515 while (!empty()) {
516 carry.splice(carry.begin(), *this, begin());
517 int i = 0;
518 while (i < fill && !counter[i].empty()) {
519 counter[i].merge(carry);
520 carry.swap(counter[i++]);
521 }
522 carry.swap(counter[i]);
523 if (i == fill) {
524 ++fill;
525 }
526 }
527 for (int i = 1; i < fill; ++i) {
528 counter[i].merge(counter[i - 1]);
529 }
530 swap(counter[fill - 1]);
531 m_size = size;
532 }

712 // 将链表变为 [first, last)
713 size_type assign_dispatch(iterator first, iterator last) {
714 size_type size = 0;
715 iterator start = begin();
716 iterator finish = end();
717 for (; start != finish && first != last; ++start, ++first, ++size) {
718 *start = *first;
719 }
720 if (first == last) {
721 erase(start, finish);
722 } else {
723 list_insert(finish, first, last);
724 }
