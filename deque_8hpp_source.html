101 self &operator+=(difference_type n) {
102 const auto offset = n + (m_cur - m_first);
103 if (offset >= 0 && offset < m_buffer_size) {

583 void m_init_map(size_type elements_count) {
584 size_type nodes_count = (elements_count / m_buffer_size) + 1;
585 m_map_size = tstl::max<size_type>(TSTL_DEQUE_MAP_INIT_SIZE, nodes_count + 2);
586 m_map = m_allocate_map(m_map_size);
587 map_pointer n_start = m_map + (m_map_size - nodes_count) / 2;
588 map_pointer n_finish = n_start + nodes_count;
589 try {
590 m_create_nodes(n_start, n_finish);
591 } catch (...) {
592 m_deallocate_map(m_map, m_map_size);
593 m_map = nullptr;
594 m_map_size = 0;
595 throw;
596 }
597 m_start.m_set_node(n_start);
598 m_finish.m_set_node(n_finish - 1);
599 m_start.m_cur = m_start.m_first;
600 m_finish.m_cur = m_finish.m_first + (elements_count % m_buffer_size);
601 }

701 template <class ForwardIt>
702 void m_range_init(ForwardIt first, ForwardIt last, tstl::forward_iterator_tag) {
703 const size_type n = tstl::distance(first, last);
704 m_init_map(n);
705 map_pointer current_node;
706 try {
707 for (current_node = m_start.m_node; current_node < m_finish.m_node; ++current_node) {
708 ForwardIt mid = first;
709 tstl::advance(mid, m_buffer_size);
710 tstl::_uninitialized_copy_a(first, mid, *current_node, m_alloc);
711 first = mid;
712 }
713 tstl::_uninitialized_copy_a(first, last, m_finish.m_first, m_alloc);
714 } catch (...) {
715 tstl::_destroy_a(m_start, iterator(*current_node, current_node), m_alloc);
716 throw;

720 iterator m_erase(iterator pos) {
721 iterator next = pos;
722 ++next;
723 const difference_type id = pos - begin();
724 if (size_type(id) < size() / 2) {
725 if (pos != begin()) {
726 tstl::move_backward(begin(), pos, next);
727 }
728 pop_front();
729 } else {
730 if (next != end()) {
731 tstl::move(next, end(), pos);
732 }
733 pop_back();
734 }
735 return begin() + id;
736 }

738 iterator m_erase(iterator first, iterator last) {
739 if (first == last) {
740 return first;
741 } else if (first == begin() && last == end()) {
742 clear();
743 return end();
744 }
745 const difference_type n = last - first;
746 const difference_type elems_before = first - begin();
747 if (size_type(elems_before) <= size() - n) {

945 const difference_type elems_before = pos - m_start;
946 const size_type length = size();
