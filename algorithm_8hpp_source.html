197// 快速排序
198template <typename RandomIt, typename Compare>
199void quick_sort(RandomIt first, RandomIt last, Compare cmp) {
200 if (first < last) {
201 typename tstl::iterator_traits<RandomIt>::value_type basic = *first;
202 RandomIt l = first;
203 RandomIt r = last - 1;
204 while (l < r) {
205 while (cmp(basic, *r) && l < r)
206 --r;
207 if (l < r) {
208 *l = *r;
209 ++l;
210 }
211 while (cmp(*l, basic) && l < r)
212 ++l;
213 if (l < r) {
214 *r = *l;
215 --r;
216 }
217 }
218 *l = basic;
219 quick_sort(first, l, cmp);
220 quick_sort(l + 1, last, cmp);
221 }
222}
